Source Code

Clean, Readable Python Code

banking_system.py
import getpass

# Initial Data
balance = 5000.00
pin = "1234"

# User Authentication
entered_pin = getpass.getpass("Enter PIN: ")

if entered_pin == pin:
  print("Login Successful!")
  while True:
    choice = input("Choice (1-4): ")
    if choice == '1':
      print(f"Balance: ₹{balance:.2f}")
    elif choice == '2':
      amount = float(input("Amount: ₹"))
      if amount > 0:
        balance += amount
    elif choice == '3':
      if amount > 20000:
        print("Limit exceeded!")
    elif choice == '4':
      break
else:
  print("Invalid PIN!")
